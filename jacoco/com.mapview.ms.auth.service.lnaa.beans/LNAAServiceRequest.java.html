package com.mapview.ms.auth.service.lnaa.beans;

import com.mapview.ms.auth.request.RequestObject;
import com.mapview.ms.auth.service.lnaa.LNAAException;
import com.mapview.ms.auth.service.lnaa.LNAASettings;

/**
 * Base class for all LNAA service requests.
 * @author cassva01
 *
 */
@SuppressWarnings("rawtypes")
public abstract class LNAAServiceRequest<T extends RequestObject> {

	public abstract LNAAServiceRequest init(LNAASettings lnaaSettings, T request, LNAARequestSession session);
	
	public abstract void validateRequest(T request) throws LNAAException;

}

package com.mapview.ms.auth.controller;

import java.util.List;

import javax.validation.constraints.NotNull;

import com.mapview.ms.auth.annotations.ValidSpecialString;
import com.mapview.ms.auth.beans.UserDto;
import com.mapview.ms.auth.entity.OauthClientDetails;
import com.mapview.ms.validators.ValidRequest;

/**
 * Auth Server Request class.
 *
 */
@ValidRequest
public class UserRequest {
	@NotNull
	@ValidSpecialString
	private String grantType;
	private List<UserDto> users;
	private List<OauthClientDetails> oauthClientDetails;

	public UserRequest() {
	}
	
	public UserRequest(String grantType, List<UserDto> users, List<OauthClientDetails> oauthClientDetails) {
		this.grantType = grantType;
		this.users = users;
		this.oauthClientDetails = oauthClientDetails;
	}

	public String getGrantType() {
		return grantType;
	}

	public void setGrantType(String grantType) {
		this.grantType = grantType;
	}

	public List<UserDto> getUsers() {
		return users;
	}

	public void setUsers(List<UserDto> users) {
		this.users = users;
	}

	public List<OauthClientDetails> getOauthClientDetails() {
		return oauthClientDetails;
	}

	public void setOauthClientDetails(List<OauthClientDetails> oauthClientDetails) {
		this.oauthClientDetails = oauthClientDetails;
	}

	@Override
	public String toString() {
		return "Request [grantType=" + grantType + ", users=" + users + ", oauthClientDetails=" + oauthClientDetails
				+ "]";
	}

}

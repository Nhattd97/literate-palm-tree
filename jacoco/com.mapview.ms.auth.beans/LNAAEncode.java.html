package com.mapview.ms.auth.beans;

import org.apache.commons.lang3.StringUtils;

public enum LNAAEncode {
    AT_RISK("@Risk", "____Risk"),
    AT_EU("@eu", "____eu"),
    AT_CHAR("@", "____"),
    DOT_CHAR(".", "___"),
    MINUS_CHAR("-", "__");

    private String value;
    private String encode;
    
    LNAAEncode(String value, String encode) {
        this.value = value;
        this.encode = encode;
    }

    public String getValue() {
        return value;
    }

    public String getEncode() {
        return encode;
    }

    public static String encode(String text) {
        for (LNAAEncode item : LNAAEncode.values()) {
            text = StringUtils.replaceIgnoreCase(text, item.getValue(), item.getEncode());
        }
        return text;
    }

    public static String decode(String text) {
        for (LNAAEncode item : LNAAEncode.values()) {
            text = StringUtils.replaceIgnoreCase(text, item.getEncode(), item.getValue());
        }
        return text;
    }

}

package com.mapview.ms.auth.beans;

import com.mapview.ms.auth.annotations.ValidSpecialString;
import com.mapview.ms.auth.annotations.ValidString;
import com.mapview.ms.auth.annotations.ValidId;
import com.mapview.ms.auth.request.RequestObject;
import lombok.AllArgsConstructor;
import lombok.Builder;
import lombok.Data;
import lombok.NoArgsConstructor;

import java.io.Serializable;
import java.util.ArrayList;
import java.util.List;

@Data
@Builder
@NoArgsConstructor
@AllArgsConstructor
@ValidId
public class UserDto extends RequestObject<UserDto> implements Serializable {
    private static final long serialVersionUID = -6967688501548567259L;

    private String username;
    
    @ValidSpecialString
    private String password;
    
    @ValidString
    private String customerNumber;
    
    @ValidString
    private String accountNumber;
    
    @ValidSpecialString
    private List<String> bundles = new ArrayList<>();
    
    @ValidSpecialString
    private String firstname;
    
    @ValidSpecialString
    private String lastname;

    private String email;
    
    @ValidSpecialString
    private String language;
    
    private String wrongLoginAttempts;
    private String expireDate;
    private String resetRequired;
    private Boolean ssoUser;
    private Long id;

    public UserDto(final String username, final String password, final String customerNumber, final String accountNumber, final List<String> bundles) {
        this.username = username;
        this.password = password;
        this.customerNumber = customerNumber;
        this.accountNumber = accountNumber;
        this.bundles = bundles;
    }
}

package com.mapview.ms.auth.service.lnaa.beans.search;

import java.util.List;

import com.fasterxml.jackson.annotation.JsonProperty;
import com.mapview.ms.auth.service.lnaa.beans.LNAAUserData;

public class LNAASearchData {

    @JsonProperty("search_counts")
	private LNAAUserSearchCounts searchCounts;
	
	private LNAAUserSearchDataPaging paging;
	
	// Change this to LNAAUserInfo for basic search
	private List<LNAAUserData> users;

	public LNAAUserSearchCounts getSearchCounts() {
		return searchCounts;
	}

	public void setSearchCounts(LNAAUserSearchCounts searchCounts) {
		this.searchCounts = searchCounts;
	}

	public LNAAUserSearchDataPaging getPaging() {
		return paging;
	}

	public void setPaging(LNAAUserSearchDataPaging paging) {
		this.paging = paging;
	}

	public List<LNAAUserData> getUsers() {
		return users;
	}

	public void setUsers(List<LNAAUserData> users) {
		this.users = users;
	}

}

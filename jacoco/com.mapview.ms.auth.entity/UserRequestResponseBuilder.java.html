package com.mapview.ms.auth.entity;

import java.util.ArrayList;
import java.util.List;
import java.util.Objects;

import org.springframework.util.CollectionUtils;

import com.mapview.ms.auth.request.RequestAccountDetails;
import com.mapview.ms.auth.response.GetUserResponse;

public class UserRequestResponseBuilder {

    private UserRequestResponse userRequestResponse;

    public UserRequestResponseBuilder() {
        userRequestResponse = new UserRequestResponse();
    }

    public UserRequestResponseBuilder(GetUserResponse userResponse) {
        try {
            UserRequestResponse requestResponse = userResponse.getUserDetails().getUser();
            userRequestResponse = Objects.isNull(requestResponse) ? new UserRequestResponse() : requestResponse;
        } catch (Exception e) {
            userRequestResponse = new UserRequestResponse();
        }
    }

    public UserRequestResponseBuilder withUsername(String username) {
        userRequestResponse.setUsername(username);
        return this;
    }

    public UserRequestResponseBuilder withFirstName(String firstName) {
        userRequestResponse.setFirstname(firstName);
        return this;
    }

    public UserRequestResponseBuilder withLastName(String lastName) {
        userRequestResponse.setLastname(lastName);
        return this;
    }

    public UserRequestResponseBuilder withEmail(String email) {
        userRequestResponse.setEmail(email);
        return this;
    }

    public UserRequestResponseBuilder withLanguage(String language) {
        userRequestResponse.setLanguage(language);
        return this;
    }

    public UserRequestResponseBuilder withBundles(List<String> bundles) {
        userRequestResponse.setBundles(bundles);
        return this;
    }

    public UserRequestResponseBuilder withRoles(List<UserRole> roles) {
        userRequestResponse.setRoles(roles);
        return this;
    }

    public UserRequestResponseBuilder withRoleBundles(List<RoleBundle> roleBundles) {
        if(!CollectionUtils.isEmpty(roleBundles)) {
            List<RoleBundle> listRoleBundles = new ArrayList<>();
            roleBundles.stream().forEach(role -> {
                    listRoleBundles.add(role);
            });  
            userRequestResponse.setRoleBundles(listRoleBundles);
        }
        return this;
    }

    public UserRequestResponseBuilder withAccountDetails(RequestAccountDetails accountDetails) {
        userRequestResponse.setAccountDetails(accountDetails);
        return this;
    }

    public UserRequestResponse build() {
        return userRequestResponse;
    }

}

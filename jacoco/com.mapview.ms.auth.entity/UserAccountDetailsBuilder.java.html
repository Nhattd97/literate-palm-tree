package com.mapview.ms.auth.entity;

import com.mapview.ms.auth.response.GetUserResponse;

import java.util.Objects;

public class UserAccountDetailsBuilder {

    private UserAccountDetails userAccountDetails;

    public UserAccountDetailsBuilder() {
        userAccountDetails = new UserAccountDetails();
    }

    public UserAccountDetailsBuilder(GetUserResponse userResponse) {
        try {
            UserAccountDetails accountDetails = userResponse.getUserDetails().getAccountDetails();
            userAccountDetails = Objects.isNull(accountDetails) ? new UserAccountDetails() : accountDetails;
        } catch (Exception e) {
            userAccountDetails = new UserAccountDetails();
        }
    }

    public UserAccountDetailsBuilder withLastLogin(String lastLogin) {
        userAccountDetails.setLastLogin(lastLogin);
        return this;
    }

    public UserAccountDetailsBuilder withStatus(String status) {
        userAccountDetails.setStatus(status);
        return this;
    }

    public UserAccountDetailsBuilder withLocked(boolean locked) {
        userAccountDetails.setLocked(locked);
        return this;
    }

    public UserAccountDetailsBuilder withPasswordExpired(boolean isExpired) {
        userAccountDetails.setExpired(isExpired);
        return this;
    }

    public UserAccountDetailsBuilder withPasswordExpireDate(String expireDate) {
        userAccountDetails.setExpireDate(expireDate);
        return this;
    }

    public UserAccountDetailsBuilder withWrongPasswordLoginAttempts(String wrongLoginAttempts) {
        userAccountDetails.setWrongLoginAttempts(wrongLoginAttempts);
        return this;
    }

    public UserAccountDetailsBuilder withResetPasswordRequired(String resetRequired) {
        userAccountDetails.setResetRequired(resetRequired);
        return this;
    }

    public UserAccountDetails build() {
        return userAccountDetails;
    }

}

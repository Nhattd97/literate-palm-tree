package com.mapview.ms.auth.entity;

public class UserDetailsBuilder {

    private UserDetails userDetails;

    public UserDetailsBuilder() {
        userDetails = new UserDetails();
    }

    public UserDetailsBuilder withUser(UserRequestResponse user) {
        userDetails.setUser(user);
        return this;
    }

    public UserDetailsBuilder withAccountDetails(UserAccountDetails accountDetails) {
        userDetails.setAccountDetails(accountDetails);
        return this;
    }

    public UserDetails build() {
        return userDetails;
    }

}

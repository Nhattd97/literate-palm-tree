package com.mapview.ms.auth.entity;

import java.util.List;

public class AccessTokenBuilder {

    private AccessToken accessToken;

    public AccessTokenBuilder() {
        accessToken = new AccessToken();
    }

    public AccessTokenBuilder withUsername(String username) {
        accessToken.setUserName(username);
        return this;
    }

    public AccessTokenBuilder withScope(List<String> scope) {
        accessToken.setScope(scope);
        return this;
    }

    public AccessTokenBuilder withBundles(List<String> bundles) {
        accessToken.setBundles(bundles);
        return this;
    }

    public AccessTokenBuilder withRoles(List<UserRole> roles) {
        accessToken.setRoles(roles);
        return this;
    }

    public AccessTokenBuilder withRoleBundles(List<RoleBundle> roleBundles) {
        accessToken.setRoleBundles(roleBundles);
        return this;
    }

    public AccessTokenBuilder withExp(String exp) {
        accessToken.setExp(exp);
        return this;
    }

    public AccessTokenBuilder withCustomerNumber(String customerNumber) {
        accessToken.setCustomerNumber(customerNumber);
        return this;
    }

    public AccessTokenBuilder withAccountNumber(String accountNumber) {
        accessToken.setAccountNumber(accountNumber);
        return this;
    }

    public AccessTokenBuilder withJti(String jti) {
        accessToken.setJti(jti);
        return this;
    }
    
    public AccessTokenBuilder withPasswordExpireDate(String passwordExpireDate) {
        accessToken.setPasswordExpireDate(passwordExpireDate);
        return this;
    }

    public AccessTokenBuilder withWrongPasswordLoginAttempts(String wrongPasswordLoginAttempts) {
        accessToken.setWrongPasswordLoginAttempts(wrongPasswordLoginAttempts);
        return this;
    }

    public AccessTokenBuilder withResetPasswordRequired(Boolean resetPasswordRequired) {
        accessToken.setResetPasswordRequired(resetPasswordRequired);
        return this;
    }

    public AccessTokenBuilder withClientId(String clientId) {
        accessToken.setClientId(clientId);
        return this;
    }
    
    public AccessTokenBuilder withKid(String kid) { 
    	accessToken.setKid(kid);
    	return this;
    }

    public AccessTokenBuilder withLanguage(String language){
        accessToken.setLanguage(language);
        return this;
    }
    public AccessToken build() {
        return accessToken;
    }

}

package com.mapview.ms.auth.entity;

import java.util.ArrayList;
import java.util.List;

import com.mapview.ms.auth.exceptions.ExceptionResponse;
import com.mapview.ms.auth.response.BaseServiceResponse;

public class UserSearchResponse extends BaseServiceResponse {

	private List<UserDetails> results = new ArrayList();
	
	private UserSearchPaging paging;
	public UserSearchResponse() {
		super(null);
	}	
	public UserSearchResponse(ExceptionResponse ex) {
		super(ex);
	}
	public List<UserDetails> getResults() {
		return results;
	}

	public void setResults(List<UserDetails> results) {
		this.results = results;
	}

	public UserSearchPaging getPaging() {
		return paging;
	}

	public void setPaging(UserSearchPaging paging) {
		this.paging = paging;
	}
	
}

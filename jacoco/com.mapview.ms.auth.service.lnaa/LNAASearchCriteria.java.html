package com.mapview.ms.auth.service.lnaa;

import java.util.List;

import com.mapview.ms.auth.request.RequestObject;

public class LNAASearchCriteria extends RequestObject {

	private List<LNAASearchField> searchFields;
	
	private String pageNumber;
	
	private String sortField;
	
	private String sortOrder;

	public List<LNAASearchField> getSearchFields() {
		return searchFields;
	}

	public void setSearchFields(List<LNAASearchField> searchFields) {
		this.searchFields = searchFields;
	}

	public String getPageNumber() {
		return pageNumber;
	}

	public void setPageNumber(String pageNumber) {
		this.pageNumber = pageNumber;
	}

	public String getSortField() {
		return sortField;
	}

	public void setSortField(String sortField) {
		this.sortField = sortField;
	}

	public String getSortOrder() {
		return sortOrder;
	}

	public void setSortOrder(String sortOrder) {
		this.sortOrder = sortOrder;
	}

}

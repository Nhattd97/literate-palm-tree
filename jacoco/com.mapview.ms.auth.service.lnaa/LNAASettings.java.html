package com.mapview.ms.auth.service.lnaa;

import org.springframework.beans.factory.annotation.Value;
import org.springframework.stereotype.Component;

/**
 * Settings for LNAA populated from the application.yml file locally
 * 
 * @author cassva01
 *
 */
@Component
public class LNAASettings {

    @Value("${mvg.lnaa.endpoint}")
    private String endpoint;
    
    @Value("${mvg.lnaa.login_id}")
    private String login_id;
    
    @Value("${mvg.lnaa.password}")
    private String password;
    
    @Value("${mvg.lnaa.timeout}")
    private Integer timeout;

    @Value("${mvg.lnaa.domain}")
    private String domain;
    
    @Value("${mvg.lnaa.user_domain}")
    private String user_domain;

    @Value("${mvg.lnaa.internal_user_domain}")
    private String internalUserDomain;
    
    @Value("${mvg.lnaa.page_size}")
    private String page_size;

    @Value("${mvg.lnaa.timeout_select_role}")
    private Integer timeoutSelectRole;
    
    public String getEndpoint() {
        return endpoint;
    }

    public String getLogin_id() {
        return login_id;
    }

    public void setPassword(String password)   {
        this.password = password;
    }
    public String getPassword() {
        return password;
    }

    public Integer getTimeout() {
        
        if(timeout != null) {
            return timeout;
        }
            return Integer.valueOf(0) ;
    }

    public String getDomain() {
        return domain;
    }

	public String getUser_domain() {
		return user_domain;
	}

	public String getPage_size() {
		return page_size;
	}

	public Integer getTimeoutSelectRole() {
		return timeoutSelectRole;
	}

	public void setTimeoutSelectRole(Integer timeoutSelectRole) {
		this.timeoutSelectRole = timeoutSelectRole;
	}

	public String getInternalUserDomain() {
		return internalUserDomain;
	}

	public void setInternalUserDomain(String internalUserDomain) {
		this.internalUserDomain = internalUserDomain;
	}

}

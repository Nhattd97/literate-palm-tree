package com.mapview.ms.auth.config.properties;

import lombok.Getter;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.stereotype.Component;

@Getter
@Component
public class DatabaseProperties {

    public static final String GLOBAL_CUSTOMER = "mapview";

    @Value("${mvg.database.driverName}")
    private String driverName;

    @Value("${mvg.database.url}")
    private String url;

    @Value("${mvg.database.dbName}")
    private String dbName;

    @Value("${mvg.database.schema}")
    private String schema;

    @Value("${mvg.database.liquibase.username}")
    private String liquibaseUsername;

    @Value("${mvg.database.liquibase.password}")
    private String liquibasePassword;

    @Value("${mvg.database.liquibase.applyAttempts}")
    private Integer liquibaseApplyAttempts;

    @Value("${mvg.database.liquibase.lockTimeout}")
    private Integer liquibaseLockTimeout;

    @Value("${mvg.database.liquibase.unlockTimeout}")
    private Integer liquibaseUnlockTimeout;

    @Value("${mvg.database.service.rolename}")
    private String serviceRolename;

    @Value("${mvg.database.service.username}")
    private String serviceUsername;

    @Value("${mvg.database.service.password}")
    private String servicePassword;

    public String getSchemaForCustomer(String customer) {
        return schema.replaceAll(GLOBAL_CUSTOMER, customer);
    }

}

package com.mapview.ms.auth.service;

import com.mapview.ms.auth.service.lnaa.UserServiceLNAA;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.context.ApplicationContext;
import org.springframework.stereotype.Service;

@Service
public class UserServiceFactoryImpl implements  UserServiceFactory {
    @Autowired
    private ApplicationContext applicationContext;

    @Override
    public UserService getUserService(String userServiceType) {
        if (UserServiceType.DB.name().equalsIgnoreCase(userServiceType)) {
            return applicationContext.getBean(UserServiceImpl.class);
        }
        if (UserServiceType.LNAA.name().equalsIgnoreCase(userServiceType)) {
            return applicationContext.getBean(UserServiceLNAA.class);
        }
        if (UserServiceType.SSO.name().equalsIgnoreCase(userServiceType)) {
            return applicationContext.getBean(UserServiceSSOImpl.class);
        }

        throw new IllegalArgumentException("Invalid user service type: " + userServiceType);
    }
}

package com.mapview.ms.auth.exceptions;


public enum AuthenticationStatus {

    UNEXPECTED_ERROR        (-2,  "MVG160"),
    LOGIN_SUCCESS           (0,   null), 
    WRONG_PASSWORD          (3,   "MVG161"), 
    ACCOUNT_LOCKED          (6,   "MVG162"),
    ACCOUNT_NOT_IN_SYSTEM   (7,   "MVG161"), 
    PASSWORD_RESET_REQUIRED (9,   "MVG163"),
    PASSWORD_EXPIRED        (10,  "MVG169"),
    ACCOUNT_DISABLED        (11,  "MVG164"),
    INVALID_BUNDLES         (12,  "MVG160");

    private  int code;
    private  String mvgCode;

    AuthenticationStatus(int code, String mvgCode) {
        this.code = code;
        this.mvgCode = mvgCode;
    }

    public int getCode() {
        return code;
    }

    public String getMvgCode() {
        return mvgCode;
    }
    
    public static String getMvgCode(int code) {
        for(AuthenticationStatus status: AuthenticationStatus.values()) {
            if(status.getCode() == code) {
                return status.getMvgCode();
            }
        }
        return UNEXPECTED_ERROR.getMvgCode();
    }
 
}

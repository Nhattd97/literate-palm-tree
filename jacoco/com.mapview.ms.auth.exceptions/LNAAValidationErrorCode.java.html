package com.mapview.ms.auth.exceptions;


public enum LNAAValidationErrorCode {
	UNEXPECTED_ERROR	(-2, "MVG160"), 
	UPDATE_SUCCESS		(0, null), 
	VALIDATION_ERROR	(5, "MVG165");

	private int code;
	private String mvgCode;

	LNAAValidationErrorCode(int code, String mvgCode) {
		this.code = code;
		this.mvgCode = mvgCode;
	}

	public int getCode() {
		return code;
	}

	public String getMvgCode() {
		return mvgCode;
	}

	public static String getMvgCode(int code) {
		for (LNAAValidationErrorCode status : LNAAValidationErrorCode.values()) {
			if (status.getCode() == code) {
				return status.getMvgCode();
			}
		}
		return UNEXPECTED_ERROR.getMvgCode();
	}

	public static LNAAValidationErrorCode getLNAAValidationErrorCode(int statusCode) {
		for (LNAAValidationErrorCode status : LNAAValidationErrorCode.values()) {
			if (status.getCode() == statusCode) {
				return status;
			}
		}
		return UNEXPECTED_ERROR;
	}
}

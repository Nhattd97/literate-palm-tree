package com.mapview.ms.auth.exceptions;

/**
 * ErrorCodes ENUM with error codes and description of error
 *
 */
public enum ErrorCode {
	
	UNAUTHORISED_PRIVILEGE("MVG001", "Not authorized"),
	BAD_REQUEST("MVG002", "Invalid request"),
	NOT_FOUND("MVG003", "Not found"),
	ERROR_VALIDATING_REQUEST("MVG005", "Invalid fields: %s"),
	LNAA_LOGIN_UNEXPECTED_ERROR("MVG160", "An unexpected error has occurred, please try again"),
	LNAA_LOGIN_WRONG_CREDENTITALS("MVG161", "The user credentials are not valid. Please try again"),
	LNAA_LOGIN_ACCOUNT_LOCKED("MVG162",  "The user credentials are not valid. Please try again"),
	LNAA_LOGIN_PASSWORD_RESET_REQUIRED("MVG163", "Your password has expired. Please set a new one"),
	LNAA_LOGIN_ACCOUNT_DISABLED("MVG164", "The user credentials are not valid. Please try again"),
	VALIDATION_ERROR("MVG165", "Validation error, Please check all fields and try again"),
	CAPTCHA_VALIDATION_ERROR("MVG166", "The user credentials are not valid. Please try again"),
	IP_VALIDATION_ERROR("MVG167", "The user credentials are not valid. Please try again"),
	IP_NOT_IN_RANGE_ERROR("MVG168", "The user credentials are not valid. Please try again"),
	LNAA_LOGIN_PASSWORD_EXPIRED("MVG169", "The user credentials are not valid. Please try again"),
	   /**
     * INTERNAL_SERVER_ERROR Please avoid using this error code unless is required
     */
    INTERNAL_SERVER_ERROR("MVG999", "ERROR: Internal server error.");
	
	private final String code;
	private final String compareString;
	
	ErrorCode(String code, String compareString) {
		this.code = code;
		this.compareString = compareString;
	}

	public String getCode() {
		return code;
	}

	public String getCompareString() {
		return compareString;
	}
	
	public static String getCompareString(String code) {
	    for(ErrorCode errorCode: ErrorCode.values()) {
	        if(errorCode.getCode().equalsIgnoreCase(code)) {
	            return errorCode.getCompareString();
	        }
	    }
	    return null;
	}
	public static ErrorCode getErrorCode(String code) {
	    for(ErrorCode errorCode: ErrorCode.values()) {
	        if(errorCode.getCode().equalsIgnoreCase(code)) {
	            return errorCode;
	        }
	    }
	    return null;
	}
}

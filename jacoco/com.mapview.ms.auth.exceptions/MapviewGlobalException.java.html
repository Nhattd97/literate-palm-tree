package com.mapview.ms.auth.exceptions;

/**
 * MapviewGlobalException - User defined exception class
 *
 */
public class MapviewGlobalException extends RuntimeException {
    
    private static final long serialVersionUID = -8599879388205641315L;
    
    private ErrorCode errorCode;
    
    public MapviewGlobalException() {}
    
    public MapviewGlobalException(String message) {
        this(ErrorCode.BAD_REQUEST, message);
    }
    
    public MapviewGlobalException(Throwable e) {
        this(ErrorCode.BAD_REQUEST, new String(), e);
    }
    
    public MapviewGlobalException(String message, Throwable e) {
        this(ErrorCode.BAD_REQUEST, message, e);
    }
    
    public MapviewGlobalException(ErrorCode errorCode, String message) {
        super(message);
        this.errorCode = errorCode;
    }
    
    public MapviewGlobalException(ErrorCode errorCode, String message, Throwable e) {
        super(message, e);
        this.errorCode = errorCode;
    }
    
    public ErrorCode getErrorCode() {
        return errorCode;
    }
}

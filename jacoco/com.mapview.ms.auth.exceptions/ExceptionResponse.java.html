package com.mapview.ms.auth.exceptions;

import java.util.Date;

import com.fasterxml.jackson.annotation.JsonFormat;

/**
 * ExceptionResponse class JSON response with exception details.
 *
 */
public class ExceptionResponse {
	
	// Using the ISO 8601 data format
	@JsonFormat
    (shape = JsonFormat.Shape.STRING, pattern = "yyyy-MM-dd hh:mm:ss")
	private Date timestamp;
	private String code;
	private String message;
	private String responseId;
	private String detailError;

    public ExceptionResponse() {
    }

	public ExceptionResponse(Date timestamp, String code, String message, String responseId) {
        this.timestamp = timestamp;
        this.code = code;
        this.message = message;
        this.responseId = responseId;
        this.detailError = message;
    }
	public ExceptionResponse(Date timestamp, String code, String message, String responseId, String detailError) {
        this.timestamp = timestamp;
        this.code = code;
        this.message = message;
        this.responseId = responseId;
        this.detailError = detailError;
    }

    public Date getTimestamp() {
        return timestamp;
    }

    public String getCode() {
        return code;
    }

    public String getMessage() {
        return message;
    }

    public String getResponseId() {
        return responseId;
    }

    public String getDetailError() {
        return detailError;
    }
}

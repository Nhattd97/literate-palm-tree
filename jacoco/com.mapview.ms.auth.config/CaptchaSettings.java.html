package com.mapview.ms.auth.config;

import org.springframework.beans.factory.annotation.Value;
import org.springframework.boot.autoconfigure.condition.ConditionalOnProperty;
import org.springframework.stereotype.Component;

@Component
@ConditionalOnProperty(prefix = "mvg", name = "google.captcha.enable", havingValue = "true", matchIfMissing = false)
public class CaptchaSettings {

	@Value("${mvg.google.captcha.site}")
	private String site;

	@Value("${mvg.google.captcha.secret}")
	private String secret;

	@Value("${mvg.google.captcha.endpoint}")
	private String endpoint;

	public CaptchaSettings() {
	}

	public String getSite() {
		return site;
	}

	public String getSecret() {
		return secret;
	}

	public String getEndpoint() {
		return endpoint;
	}
}

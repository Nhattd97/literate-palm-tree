package com.mapview.ms.auth.request;

import com.mapview.ms.auth.response.GetUserResponse;

import java.util.Objects;

public class RequestAccountDetailsBuilder {

    private RequestAccountDetails requestAccountDetails;

    public RequestAccountDetailsBuilder() {
        requestAccountDetails = new RequestAccountDetails();
    }

    public RequestAccountDetailsBuilder(GetUserResponse response) {
        try {
            RequestAccountDetails accountDetails = response.getUserDetails().getUser().getAccountDetails();
            requestAccountDetails = Objects.isNull(accountDetails) ? new RequestAccountDetails() : accountDetails;
        } catch (Exception e) {
            requestAccountDetails = new RequestAccountDetails();
        }
    }

    public RequestAccountDetailsBuilder withCustomerNumber(String customerNumber) {
        requestAccountDetails.setRequestCust(customerNumber);
        return this;
    }

    public RequestAccountDetailsBuilder withAccountNumber(String accountNumber) {
        requestAccountDetails.setRequestAccount(accountNumber);
        return this;
    }

    public RequestAccountDetails build() {
        return requestAccountDetails;
    }

}

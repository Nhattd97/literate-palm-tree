package com.mapview.ms.auth.request;

public class RequestAccountDetails {

	private String requestCust;

	private String requestAccount;
	
	public RequestAccountDetails()	{
		
	}
	
	public RequestAccountDetails(String customer, String account) {
		this.requestCust = customer;
		this.requestAccount = account;
	}

	public String getRequestCust() {
		return requestCust;
	}

	public void setRequestCust(String requestCust) {
		this.requestCust = requestCust;
	}

	public String getRequestAccount() {
		return requestAccount;
	}

	public void setRequestAccount(String requestAccount) {
		this.requestAccount = requestAccount;
	}

}

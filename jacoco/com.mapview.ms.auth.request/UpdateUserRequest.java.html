package com.mapview.ms.auth.request;

import java.io.Serializable;
import java.util.List;

import com.mapview.ms.auth.annotations.ValidSpecialString;
import com.mapview.ms.auth.annotations.ValidUsername;

import lombok.AllArgsConstructor;
import lombok.Builder;
import lombok.Data;
import lombok.NoArgsConstructor;

@Data
@NoArgsConstructor
@AllArgsConstructor
@Builder
public class UpdateUserRequest extends RequestObject<UpdateUserRequest> implements Serializable {
	private static final long serialVersionUID = 1L;

	@ValidSpecialString
	private String username;
	
	@ValidSpecialString
	private String firstname;

	@ValidSpecialString
	private String lastname;

	@ValidSpecialString
	private String email;
	
	@ValidSpecialString
	private List<String> bundles;
	
	@ValidSpecialString
	private String language;

	private Boolean ssoUser;

	@ValidSpecialString
	private String ssoId;

	private Long id;

	public String getBundleString() {
		if(bundles == null || bundles.size() == 0) {
			return "";
		}
		String[] bds = bundles.stream().toArray(String[]::new);
		return String.join(",", bds);
	}
	
}

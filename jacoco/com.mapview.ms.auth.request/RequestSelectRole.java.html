package com.mapview.ms.auth.request;

import javax.validation.constraints.NotNull;

import com.mapview.ms.auth.annotations.ValidSpecialString;

public class RequestSelectRole {

    @NotNull
    @ValidSpecialString
    private String role;

    @ValidSpecialString
    private String scope;

    public String getRole() {
        return role;
    }

    public void setRole(String role) {
        this.role = role;
    }

    public String getScope() {
        return scope;
    }

    public void setScope(String scope) {
        this.scope = scope;
    }

    @Override
    public String toString() {
        return "RequestSelectRole{" +
            "role='" + role + '\'' +
            ", scope='" + scope + '\'' +
            '}';
    }
    
}
